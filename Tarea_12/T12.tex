\noindent \textbf{a)} Sea $f_X(x) = 1/2$. La función generadora de momentos esta dada por:
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
  g(t) &=& \int^2_0 e^{tx}\left(\frac{1}{2}\right) dx \\
   &=& \frac{1}{2} \int^2_0 e^{tx} dx \\
   &=& \frac{1}{2} \left[ \frac{e^{tx}}{t}\right]^2_0  \\

\noindent \textbf{b)} Sea $f_X(x) = (1/2)x$. La función generadora de momentos esta dada por:
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
  g(t) &=& \int^2_0 e^{tx}\left(\frac{1}{2}x\right) dx \\
   &=& \frac{1}{2} \int^2_0 x e^{tx} dx \\
   &=& \frac{1}{2} \left.\left[ \frac{x e^{tx}}{t}\right|^2_0 - \int^2_0 \frac{e^{tx}}{t} dx\right]  \\
   &=& \frac{1}{2} \left[ \frac{x e^{tx}}{t} -\frac{1}{t} \left(\frac{e^{tx}}{t}\right)\right]^2_0 \\
   &=& \frac{1}{2t} \left[\left(2e^{2t} - \frac{e^{2t}}{t}\right) - \left(0e^0 - \frac{e^0}{t}\right)\right]\\
   &=& \frac{1}{2t} \left[2e^{2t} - \frac{e^{2t}}{t} + \frac{1}{t}\right] \\
   &=& \frac{2te^{2t} - e^{2t} + 1}{2t^2}.
\end{eqnarray*}

Sea la función característica $k_X (\tau) = e^{-|\tau|}$ con $-\infty < \tau < \infty$, por demostrar que: $f_X (x) = \frac{1}{\pi (1 + x^2)}$. Luego:
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
  f_X (x) &=& \frac{1}{2\pi} \int^\infty_{-\infty} e^{-i\tau x} e^{-| \tau |} d\tau \\
   &=& \frac{1}{2\pi} \left[\int^0_{-\infty} e^{-i\tau x + \tau} d\tau + \int^\infty_0 e^{-i\tau x - \tau} d\tau\right] \\
   &=& \frac{1}{2\pi} \left[\int^0_{-\infty} e^{\tau(1-i x)} d\tau + \int^\infty_0 e^{-\tau(ix + 1)} d\tau\right] \\
   &=& \frac{1}{2\pi} \left[\left.\lim_{a\rightarrow -\infty} \frac{e^{\tau (1-ix)}}{1-ix}\right|^0_a - \left.\lim_{b\rightarrow \infty} \frac{e^{- \tau (ix + 1)}}{ix + 1}\right|^b_0\right] \\
   &=& \frac{1}{2\pi} \left[\lim_{a\rightarrow -\infty}\left(\frac{e^{0(1-ix)}}{1-ix} - \frac{e^{a(1-ix)}}{1-ix}\right) - \lim_{b\rightarrow \infty}\left(\frac{e^{-b(ix+1)}}{ix+1} - \frac{e^{0(ix+1)}}{ix+1}\right)\right] \\
   &=& \frac{1}{2\pi} \left[\frac{1}{1-ix} + \frac{1}{ix+1}\right] \\
   &=& \frac{1}{2\pi} \left(\frac{ix+1+1-ix}{(1-ix)(1+ix)}\right) \\
   &=& \frac{1}{2\pi} \left(\frac{2}{ix+1-i^2 x^2 -ix}\right) \\
   &=& \frac{1}{\pi} \left(\frac{1}{1 -(\sqrt{-1})^2 x^2}\right) \\
   &=& \frac{1}{\pi}\left(\frac{1}{1 + x^2}\right).
\end{eqnarray*}

\noindent \textbf{Ejercicio 10 -- pág. 403}\\

Sea la función de densidad $f(x) = (1/2) e^{-|x|}$ con $-\infty < x < \infty$.\\

\noindent \textbf{a)} Para poder determinar la media y la varianza de la función, se calcula la función generadora de momentos como sigue:
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
  M(t) &=& \int^\infty_{-\infty} e^{tx} \left(\frac{1}{2} e^{-|x|}\right) dx \\
   &=& \frac{1}{2} \left[\int^0_{-\infty} e^{-(-x)}e^{tx} dx + \int^\infty_0 e^{-x}e^{tx} dx\right] \\
   &=& \frac{1}{2} \left[\int^0_{-\infty} e^{x(1+t)} dx + \int^\infty_0 e^{-x(1-t)} dx\right] \\
   &=& \frac{1}{2} \left[\lim_{a\rightarrow - \infty} \left.\frac{e^{x(t+1)}}{t+1}\right|_0^a - \lim_{b\rightarrow \infty} \left.\frac{e^{-x(1-t)}}{1-t}\right|_0^b \right] \\
   &=& \frac{1}{2} \left[\lim_{a\rightarrow - \infty} \left(\frac{e^{0(t+1)}}{t+1} - \frac{e^{a(t+1)}}{t+1}\right) - \lim_{b\rightarrow \infty} \left(\frac{e^{-b(1-t)}}{1-t} - \frac{e^{-0(1-t)}}{1-t}\right) \right] \\
   &=& \frac{1}{2} \left[\frac{1}{t+1} + \frac{1}{1+t}\right] \\
   &=& \frac{1}{2} \left[\frac{1-t+t+1}{(t+1)(1-t)}\right] \\
   &=& \frac{1}{2} \left[\frac{2}{t-t^2+1-t}\right] \\
   &=& \frac{1}{1-t^2},
\end{eqnarray*}
\noindent luego, por definición:
\begin{equation*}
  M(t) = 1 + \mu t + \mu_2^{\prime} \frac{t^2}{2!} + \ldots + \mu_r^{\prime} \frac{t^r}{r!} + \ldots,
\end{equation*}
\begin{equation*}
  \mu_r^{\prime} = \left.\frac{d^r}{dt^r} M(t)\right|_{t=0},
\end{equation*}
\noindent de donde se sabe que: $\mu_1^{\prime} = \mu$ y $\mu_2 = \mu_2^{\prime} - \mu^2 = \sigma^2$. Entonces:
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
  \mu_1^{\prime} &=& \left.\frac{(1-t^2)(0) - 1(-2t)}{(1-t^2)^2}\right|_{t=0} \\
   &=& \left.\frac{2t}{(1-t^2)^2} \right|_{t=0} \\
   &=& 0 = \mu,
\end{eqnarray*}
\begin{eqnarray*}
% \nonumber to remove numbering (before each equation)
